class User(models.Model):

- 등록 URI 작성
>views.py
def register(request, user_id):
    try:
        token = request.post['token']
        User(token=token).save()
    except KeyError:
        return HttpResponse(status=404)
        
>urls.py
url(r'^/register/$',views.register, name='register')
